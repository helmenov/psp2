import pandera as pa
from pandera.typing import Series
from pandera import Field, DataFrameModel

class IrisSchema(DataFrameModel):
    sepal_length: Series[float] = Field(gt=0, le=8)
    sepal_width: Series[float] = Field(gt=0, le=5)
    petal_length: Series[float] = Field(gt=0, le=7)
    petal_width: Series[float] = Field(gt=0, le=3)
    target: Series[int] = Field(isin=[0,1,2])
iris = IrisSchema.validate(iris)
